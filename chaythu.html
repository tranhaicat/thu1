import numpy as np
def BM(T,N,nsim): 
    dt = T/N
    Z = np.random.randn(nsim,N)
    W = np.zeros((nsim,N+1))
    for t in range(0,N):
        W[:,t+1] = W[:,t] + np.sqrt(dt)*Z[:,t]
    return W
